from core.models import Usuario
from random import randint
import faker
from datetime import datetime, timedelta
import random

# Crear una instancia de Faker para generar datos aleatorios
fake = faker.Faker()

# Generar una fecha aleatoria entre dos fechas
def random_date(start_date, end_date):
    delta = end_date - start_date
    random_days = random.randint(0, delta.days)
    return start_date + timedelta(days=random_days)

# Rango de fechas para el campo fecha_registro
start_date = datetime(2022, 1, 1)
end_date = datetime.now()

# Crear 10 usuarios de tipo "Comprador" con fecha_registro aleatoria
for _ in range(100):
    usuario = Usuario.objects.create_user(
        correo_electronico=fake.email(),
        nombre=fake.name(),
        contrasena='Comprador123',  # Contrase√±a predeterminada
        tipo_usuario='Comprador',
        telefono='1234785263'
    )
    # Asignar fecha_registro aleatoria
    usuario.fecha_registro = random_date(start_date, end_date)
    usuario.save()
    print(f"Usuario {usuario.nombre} creado exitosamente con fecha de registro {usuario.fecha_registro}.")
